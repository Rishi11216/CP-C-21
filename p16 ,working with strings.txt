Program 16 : Working with strings

#include <stdio.h>
#include <string.h>
struct student
{
int eng;
int maths;
int science;
};
int main()
{
int total;
struct student stud;
stud.eng = 75;
stud.maths = 70;
stud.science = 65;
total = stud.eng + stud.maths + stud.science;
printf ("Total is %d\n",total);
return 0;
}

Output :
Total is 210
