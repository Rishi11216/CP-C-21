Program 14 : Strings

#include <stdio.h>
#include <string.h>
int main()
{
    char strname[30];
    printf("Enter the string\n");
    scanf("%[^\n]s",strname);
    printf("The string is %s\n",strname);
    return 0;
}

Output :
Enter the string
Talk To a Teacher
The string is Talk To a Teacher
